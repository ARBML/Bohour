أَرَقٌ عَلى أَرَقٍ وَمِثلِيَ يَأرَقُ
أرقن على أرقن ومثلي يأرقو

وَجَوىً يَزيدُ وَعَبرَةٌ تَتَرَقرَقُ
وجون يزيد وعبرتن تترقرقو

حَيِّ الوُجوهَ المِلاحا
حيي لوجوه لملاحا

وَحَيِّ نُجلَ العُيون
وحيي نجل لعيوني

هَل في الهَوى مِن جُناحٍ
هل فلهوى من جناحن

أَو في نَديمٍ وَراح
أو في نديمن وراحي

بَكى بِشَجوي وَناحا
بكى بشجوي وناحا

عَلى فُروعِ الغُصونِ
على فروع لغصوني

وَقَد لَهَوتُ بِمِثلِ الرِئمِ آنِسَةٍ
وقد لهوت بمثل ررئم ءانستن

تُصبي الحَليمَ عَروبٍ غَيرِ مِكلاحِ
تصبلحليم عروبن غير مكلاحي

هَبَّت جَنوبٌ بِأَعلاهُ وَمالَ بِهِ
هببت جنوبن بأعلاهو ومال بهي

أَعجازُ مُزنٍ يَسُحُّ الماءِ دَلّاحِ
أعجاز مزنن يسحح الماء دللاحي

قَد تَقَطَرنَ بالعبيرِ ومَسكٍ
قد تقططرن بلعبير ومسكن

وَتَكَبَينَ بالكباءِ ذكيا
وتكببين بلكباء ذكييا

كُنتُ دَلّالَها وَكانَ سِناني
كنت دللالها وكان سناني

تاجِراً يَشتَري النُفوسَ الغَوالي
تاجرن يشترننفوس لغوالي

ثُمَّ عودي مِن بَعدِ ذا وَاِشكُريني
ثمم عودي من بعد ذا وشكريني

وَاِذكُري ما رَأَيتِهِ مِن فِعالي
وذكري ما رأيتهي من فعالي

أَيُّها الساعي عَلى آثارِنا
أييهسساعي على ءاثارنا

نَحنُ مَن لَستَ بِسَعّاءٍ مَعَه
نحن من لست بسععاءن معه

ثُمَّ فينا لِلقِرى نارٌ يُرى
ثمم فينا للقرى نارن يرى

عِندَها لِلضَيفِ رُحبٌ وَسِعَه
عندها لضضيف رحبن وسعه

بأبي مَن لِسَبته نَحلةٌ
بأبي من لسبتهو نحلتن

آلمت أَكرَم شَيءٍ وَأَجلّ
ءالمت أكرم شيءن وأجلل

حَسِبَت أَنَّ بفيه بَيتَها
حسبت أنن بفيهي بيتها

إِذ رأَت ريقَتَهُ مثلَ العَسَل
إذ رأت ريقتهو مثل لعسلي

لقد كنتُ ذَا بَأسٍ شَدِيدٍ وَهِمَّةٍ
لقد كنت ذا بأسن شديدن وهممتن

إِذَا شِئتُ لَمساً لِلثُّريَّا لَمَستُهَا
إذا شئت لمسن لثثريا لمستها

أَتَتنِي سِهَامٌ مِن لِحَاظٍ فَأرشَقَت
أتتني سهامن من لحاظن فأرشقت

بِقَلبِيِ وَلَو أَسطِيعُ رَدّاً رَدَدَتُّهَا
بقلبي ولو أسطيع رددن ردتتها

خَليلَيَّ مُرّا بي عَلى أُمِّ جُندَبِ
خليليي مررا بي على أمم جندبي

نُقَضِّ لُباناتِ الفُؤادِ المُعَذَّبِ
نقضض لبانات لفؤاد لمعذذبي

أَلَم تَرَياني كُلَّما جِئتُ طارِقاً
ألم ترياني كللما جئت طارقن

وَجَدتُ بِها طيباً وَإِن لَم تُطَيَّبِ
وجدت بها طيبن وإن لم تطييبي

بَكَى فَرثَت له أَجبالُ صُبحٍ
بكى فرثت لهو أجبال صبحن

وأسعَدتِ الجِبالَ بها مُرُوتُ
وأسعدت لجبال بها مروتو

كأَنَّ فؤادَه كَفَّا غَرِيقٍ
كأنن فؤادهو كففا غريقن

تَنازَعَهُ بِشَطِّ البَحرِ حُوتُ
تنازعهو بشطط لبحر حوتو

أَمُساوِرٌ أَم قَرنُ شَمسٍ هَذا
أمسارون أم قرن شمسن هاذا

أَم لَيثُ غابٍ يَقدُمُ الأُستاذا
أم ليث غابن يقدم لأستاذا

غادَرتَ أَوجُهَهُم بِحَيثُ لَقِيتَهُم
غادرت أوجههم بحيث لقيتهم

أَقفائَهُم وَكُبودَهُم أَفَلاذا
أقفائهم وكبودهم أفلاذا

في مَوقِفٍ وَقَفَ الحِمامُ عَلَيهِمِ
في موقفن وقف لحمام عليهمو

في ضَنكِهِ وَاِستَحوَذَ اِستِحواذا
في ضنكهي وستحوذ ستحواذا

بَردُ نَسيمِ الحِجازِ في السَحَرِ
برد نسيم لحجاز فسسحري

إِذا أَتاني بِريحِهِ العَطِرِ
إذا أتاني بريحه لعطري

وَمُلكُ كِسرى لا أَشتَهيهِ إِذا
وملك كسرى لا أشتهيه إذا

ما غابَ وَجهُ الحَبيبِ عَن نَظَري
ما غاب وجه لحبيب عن نظري

أَرى لِلصّبا وداعا
أرى لصصبا وداعا

وما يَذْكُرُ اجْتماعا
وما يذكر جتماعا

فَجَدِّدْ وِصالَ صَبٍّ
فجددد وصال صبن

متَى تَعْصِهِ أَطاعا
متى تعصهي أطاعا

إنْ تَدْنُ مِنْهُ شِبراً
إن تدن منه شبرن

يُقَرّبْكَ مِنْهُ باعا
يقربك منه باعا

زان طرفها الحور
زان طرفهلحورو

زاغ نحوها البصر
زاغ نحوهلبصرو

يال أعين فتكت
يال أعينن فتكت

راش سهمها النظر
راش سهمهننظرو

واحمرار وجنتها
وحمرار وجنتها

الرحيق معتصر
أررحيق معتصرو

عَفا مِن آلِ لَيلى السَهـ
عفا من ءال ليلسسه

ـبُ فَالأَملاحُ فَالغَمرُ
ب فلأملاح فلغمرو

كَما قالَت فَتاةُ الحَيَّ لَمّا
كما قالت فتاة لحيي لمما

أَحَسَّ جَنانُها جَيشاً لُهاما
أحسس جنانها جيشن لهاما

أَلَستَ تَرى القَطا مُتَواتِراتٍ
ألست ترلقطا متواتراتن

وَلَو تُرِكَ القَطا أَغفى وَناما
ولو ترك لقطا أغفى وناما

إِنَّ بِالشّعبِ الذي دونَ سلعٍ
إنن بششعب للذي دون سلعن

لَقتيلاً دَمُه ما يُطَلُّ
لقتيلن دمهو ما يطللو

خَلَّفَ العِبءَ علَيَّ وَوَلّى
خللف لعبء عليين ووللى

أَنَا بِالعِبءِ له مُستَقِلُّ
أنا بلعبء لهو مستقللو

وَوَراءَ الثَّأرِ منّي ابنُ أُختٍ
ووراء ثثأر مننبن أختن

مَصِعٌ عُقدَتُهُ ما تُحَلُّ
مصعن عقدتهو ما تحللو

تَطاوَلَ لَيلُكَ بِالأَثمَدِ
تطاول ليلك بلأثمدي

وَنامَ الخَلِيُّ وَلَم تَرقُدِ
ونام لخليي ولم ترقدي

وَباتَ وَباتَت لَهُ لَيلَةٌ
وبات وباتت لهو ليلتن

كَلَيلَةِ ذي العائِرِ الأَرمَدِ
كليلة ذلعائر لأرمدي

مَن ذا أَفتاكِ بِسَفكِ دَمي
من ذا أفتاك بسفك دمي

يا غُرَّةَ حَيِّ بَني جُشَمِ
يا غررة حيي بني جشمي

أَبِنَظرَةِ عَينٍ عَن خَطأٍ
أبنظرة عينن عن خطئن

عَرَضَت بِالعَمدِ يُراقُ دَمي
عرضت بلعمد يراق دمي

يا أَيُّها المُزمِعُ ثُمَّ اِنثَنى
يا أييهلمزمع ثمم نثنى

لا يَثنِكَ الحازي وَلا الشاحِجُ
لا يثنك لحازي وششاحجو

ولا قَعيدٌ أَغَضَبٌ قَرنُهُ
ولا قعيدن أغضبن قرنهو

هاجَ لَهُ مِن مَرتَعٍ هائِجُ
هاج لهو من مرتعن هائجو

قَد كُنتَ يَوماً تَرتَجي رِسلَها
قد كنت يومن ترتجي رسلها

فَأَطرِدَ الحائِلُ وَالدالِجُ
فأطرد لحائل وددالجو

قُلتُ مَنِ القَومُ فَقالوا سَفَرَه
قلت من لقوم فقالو سفره

وَالقَومُ كَعبٌ يَبتَغونَ المُنكَرَه
ولقوم كعبن يبتغون لمنكره

قُلتُ لِكَعبٍ وَالقَنا مُشتَجِرَه
قلت لكعبن ولقنا مشتجره

تَعَلَّمي يا كَعبُ وَاِمشي مُبصِرَه
تعللمي يا كعب ومشي مبصره
